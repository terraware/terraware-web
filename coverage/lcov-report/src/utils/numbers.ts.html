  // Allow control keys
  if (['Backspace', 'Delete', 'Tab', 'Escape', 'Enter', 'ArrowLeft', 'ArrowRight'].includes(key)) {
    return;
  }

  // If decimal separator is entered
  if (key === '.' || key === ',') {
    // Prevent if already has a decimal separator
    if (value.includes('.') || value.includes(',')) {
